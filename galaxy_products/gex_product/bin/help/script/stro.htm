strSubString(
  const string,     // source string
  const start,      // starting position, index based 1
  [const count,     // # of characters
   const padchar])  // padding character

Returns a sub-string from a string. If count is omitted, all remaining characters are taken.

Examples:

const s = 'hello world';

substr = strSubString(s,2,2);         // substr = 'el'
substr = strSubString(s,4);           // substr = 'lo world'
substr = strSubString(s,1,20,'*');    // substr = hello world*********
substr = strSubString('',1,20,'-');   // substr = --------------------
